\begin{exercise}
Consider the differential equation $\frac{d^2y}{dx^2} +9y =0$.

What is meant by a solution to this equation?

\begin{multipleChoice}
\choice{The solution is an $x$-value for which the above relationship holds.}
\choice[correct]{The solution is a function $y(x)$ for which the above relationship holds for all values of $x$ in its domain.}
\end{multipleChoice}



Select all of the functions below that are solutions to this equation.

\begin{selectAll}
\choice{$y = 5$}
\choice{$y = 3e^{3x}$}
\choice[correct]{$y = 6\cos(3x)$}
\choice{$y = 5\sin(9x)$}
\choice[correct]{$y = -6\sin(3x)$}
\choice[correct]{$y = C_1 \cos(3x)+C_2\sin(3x)$, where $C_1$ and $C_2$ are arbitrary constants.}
\end{selectAll}


\end{exercise}
