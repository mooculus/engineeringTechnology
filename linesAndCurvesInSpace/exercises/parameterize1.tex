\begin{exercise}
Which of the following is a parameterization of the circle $x^2+y^2=25$?

\begin{selectAll}
\choice{$\vec{p}(t) = \vector{25 \cos(t),25\sin(t)}, 0 \leq t \leq 2\pi$}
\choice[correct]{$\vec{p}(t) = \vector{5 \cos(t),5\sin(t)}, 0 \leq t \leq 2\pi$}
\choice[correct]{$\vec{p}(t) = \vector{5 \sin(2t),5\cos(2t)}, 0 \leq t \leq \pi$}
\choice{$\vec{p}(t) = \vector{t,\sqrt{25-t^2}}, -5\leq t \leq 5$}
\choice{$\vec{p}(t) =  \vector{t,-\sqrt{25-t^2}}, -5\leq t \leq 5$}
\end{selectAll} 
\end{exercise}
