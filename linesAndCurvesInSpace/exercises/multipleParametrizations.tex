\outcome{Given a curve, identify multiple parametrizations.}

\begin{document}

\begin{exercise}

  Consider the curve below.
  \begin{image}
    \begin{tikzpicture}
      \draw[->] (-1,0) -- (1,0);
      \draw[->] (0,-0.1) -- (0,1);
      \draw[domain=-1:1,smooth,variable=\x,penColor,ultra thick]  plot ({\x},{\x*\x});
    \end{tikzpicture}
  \end{image}

  Which of the following could be parametrization of the above curve?
  \begin{selectAll}
    \choice[correct]{$\vec{v}(t) = \vector{\sin t,\sin^2 t}$ for $t \in \R$.}
    \choice{$\vec{v}(t) = \vector{\sqrt{t},t}$ for $t \in [0,1]$.}
    \choice[correct]{$\vec{v}(t) = \vector{\cos t,\cos^2 t}$ for $t \in \R$.}
    \choice[correct]{$\vec{v}(t) = \vector{t^3,t^6}$ for $t \in [-1,1]$.}
    \choice{$\vec{v}(t) = \vector{t^2,t}$ for $t \in [-1,1]$.}
    \choice[correct]{$\vec{v}(t) = \vector{t,t^2}$ for $t \in [-1,1]$.}    
  \end{selectAll}

\end{exercise}
