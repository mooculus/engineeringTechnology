\begin{exercise}
Which of the following is a parameterization of the ellipse $16x^2+9y^2=144$?

\begin{selectAll}
\choice[correct]{$\vec{p}(t) = \vector{3 \cos(t),4\sin(t)}, 0 \leq t \leq 2\pi$}
\choice{$\vec{p}(t) = \vector{4 \cos(t),3\sin(t)}, 0 \leq t \leq 2\pi$}
\choice[correct]{$\vec{p}(t) = \vector{3 \sin(2t),4\cos(2t)}, 0 \leq t \leq \pi$}
\choice{$\vec{p}(t) = \vector{t,\frac{1}{3}\sqrt{144-16t^2}}, -3\leq t \leq 3$}
\choice{$\vec{p}(t) = \vector{t,-\frac{1}{3}\sqrt{144-16t^2}}, -3\leq t \leq 3$}
\end{selectAll} 
\end{exercise}
