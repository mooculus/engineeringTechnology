\begin{exercise}

\[
\int \frac{\cos(\frac{1}{x})}{x^2} \d x \begin{prompt} = \answer{-\sin\left(\frac{1}{x}\right)}+C \end{prompt}
\]
\begin{hint}
Use: $u=\frac{1}{x}$.

\end{hint}
\begin{hint}
Then: $\d u=-\frac{1}{x^2}\d x$,  $\d x=-(x^2)\d u$, and 

$\int \frac{\cos(\frac{1}{x})}{x^2} \d x=\int \frac{\cos(u)}{x^2}(-x^2)\d u=-\int \cos(u)\d u$

\end{hint}
\end{exercise}
