\begin{exercise}

Find the antiderivative: \[\int 2e^{4x}\; dx.\]


\begin{hint}
Use: $u=4x$.
\end{hint}


\begin{hint}
Then $\d u=4\d x$, $\d x=\frac{1}{4}\d u$, and 
\[\int 2e^{4x}\; dx=\int 2e^{u}\frac{1}{4}\d u\;.\]
\end{hint}


\begin{multipleChoice}
\choice{$2e^{4x}+C$}
\choice{$8e^{4x}+C$}
\choice{$8e^{4x}$}
\choice{$2e^{4x}$}
\choice[correct]{$\frac{1}{2}e^{4x}+C$}
\end{multipleChoice}

\end{exercise}
