\begin{exercise}

Evaluate the following definite integral of $(x+1)e^{x^2+2x+1}$ with respect to $x$:

\[
\int_{-2}^{-1}(x+1)e^{x^2+2x+1}\d x=\answer{\frac{1-e}{2}}
\]

\begin{hint}
Use: $u=x^2+2x+1$.
\end{hint}
\begin{hint}
Then,  $\d u=(2x+2)\d x=2(x+1)\d x$.
\end{hint}
\end{exercise}
