\begin{exercise}
  Let $C$ be the curve drawn from $A$ to $B$ below:
  \begin{image}
    \begin{tikzpicture}
      \begin{axis}%
        [
	  xmin=-7,xmax=7,
          ymin=-5,ymax=5,
          xlabel=$x$,ylabel=$y$,
          axis lines=center,
          every axis y label/.style={at=(current axis.above origin),anchor=south},
          every axis x label/.style={at=(current axis.right of origin),anchor=west},
          clip=false,
	  grid =major,
          xtick={-7,-6,...,7},
          ytick={-5,-4,...,5},
	]
        \addplot[penColor,ultra thick,smooth] coordinates{
          (-5,3) (4,4)  (-5,-2) (4,-3)};
        \addplot[color=penColor,fill=penColor,only marks,mark=*] coordinates{(-5,3)};  %% closed hole
        \addplot[color=penColor,fill=penColor,only marks,mark=*] coordinates{(4,-3)};  %% closed hole
        \node[left] at (axis cs:-5,3) {$A$};
        \node[right] at (axis cs:4,-3) {$B$};
      \end{axis}
    \end{tikzpicture}
  \end{image}
  Compute:
  \[
  \int_C x\d x + (4+2y)\d y = \answer{-57/2}
  \]
\end{exercise}
