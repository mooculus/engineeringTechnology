\begin{exercise}

Let $g$ be a function such that 
\begin{itemize}
\item $g(2) = 7$,
\item $g(5) = 1$, and
\item $g'(x) < 0$ for all $x>0$.
\end{itemize}

Chose all correct statements below.
\begin{selectAll}
\choice{$g$ does \textbf{not} satisfy the conditions of the Mean Value Theorem.}
\choice[correct]{$g'(c) = -2$ for some number $c$ in $(2,5)$.}
\choice{$-2<g'(x)<0$ for all $x$ in $(2,5)$.}
\choice[correct]{$1\leq g(x)\leq7$ for all $x$ in $(2,5)$.}
\end{selectAll}

\end{exercise}
