Select all functions that satisfy the conditions of the Mean Value Theorem on $[1,5]$.
\begin{selectAll}
\choice{$f_A(x)$}
\choice{$f_B(x)$}
\choice[correct]{$f_C(x)$}
\choice{$f_D(x)$}
\end{selectAll}

Select all functions $f$ for which there exists a point $c$ in $(1,5)$ such that
\[
f'(c) = \frac{f(5)-f(1)}{5-1}.
\]
\begin{selectAll}
\choice[correct]{$f_A(x)$}
\choice{$f_B(x)$}
\choice[correct]{$f_C(x)$}
\choice{$f_D(x)$}
\end{selectAll}
