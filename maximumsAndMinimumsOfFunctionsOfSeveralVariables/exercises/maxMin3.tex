\begin{exercise}
  Find the critical points of $F(x,y) = x^2 + 3 y^2 - 6 y + 4 x y$.
  \[
  \vec{c} = \vector{\answer{6},\answer{-3}}
  \]
  \begin{exercise}
  \[
  D(\vec{c}) = \answer{-4}
  \]
  \begin{exercise}
    At $\vec{c}$, $F$ locally looks like:
    \begin{multipleChoice}
      \choice{an elliptic paraboloid}
      \choice[correct]{a hyperbolic paraboloid}
      \choice{neither of these}
    \end{multipleChoice}
    \begin{exercise}
      The point $\vec{c}$ is:
      \begin{multipleChoice}
        \choice{a local minimum for $F$}
        \choice{a local maximum for $F$}
        \choice[correct]{neither a local minimum nor a local maximum for $F$}
      \end{multipleChoice}
    \end{exercise}
  \end{exercise}
  \end{exercise}
\end{exercise}
