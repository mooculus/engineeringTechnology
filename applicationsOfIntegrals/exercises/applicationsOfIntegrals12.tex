Select the expression that is equal to $\dd{x}\left[\int_0^{\cos x} \tan\left(t^3-1\right)\d t\right]$.
\begin{multipleChoice}
\choice{$\tan\left(x^3-1\right)$}
\choice{$\tan\left(\cos^3(x)-1\right)$}
\choice{$3x^2\sec^2\left(x^3-1\right)$}
\choice{$\tan\left(\cos^3(x)-1\right) + C$}
\choice[correct]{$-\sin(x)\tan\left(\cos^3(x)-1\right)$}
\choice{$3\cos^2(x)\sec^2\left(\cos^3(x)-1\right)$}
\choice{None of the above answers}
\end{multipleChoice}
