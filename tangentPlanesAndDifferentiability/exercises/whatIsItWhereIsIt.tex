\begin{exercise}
Let $f:\R^2 \to \R$ and let $z=f(x,y)$ be the surface associated to the function.

I. Classify the following as scalars, vectors in $\R^2$, or vectors in $\R^3$.

\begin{itemize}
\item $f(3,5)$ is a \wordChoice{\choice[correct]{scalar}\choice{vector in $\R^2$}\choice{vector in $\R^3$}}.
\item $\grad{f}(3,5)$ is a \wordChoice{\choice{scalar}\choice[correct]{vector in $\R^2$}\choice{vector in $\R^3$}}. 
\item $f_y(3,5)$ is a \wordChoice{\choice[correct]{scalar}\choice{vector in $\R^2$}\choice{vector in $\R^3$}}.
\item The normal vector to the tangent plane to the surface at $(3,5,f(3,5))$ is a \wordChoice{\choice{scalar}\choice{vector in $\R^2$}\choice[correct]{vector in $\R^3$}}.
\end{itemize}

II. Classify the following as subsets of $\R$, $\R^2$, or $\R^3$.

\begin{itemize} 
\item The level curve of $f(x,y)$ through $(3,5)$ is a subset of \wordChoice{\choice{$\R$}\choice[correct]{$\R^2$}\choice{$\R^3$}}.
\item The domain of $f(x,y)$ is a subset of \wordChoice{\choice{$\R$}\choice[correct]{$\R^2$}\choice{$\R^3$}}.
\item The range of $f(x,y)$  is a subset of \wordChoice{\choice[correct]{$\R$}\choice{$\R^2$}\choice{$\R^3$}}.
\item The surface $z=f(x,y)$  is a subset of \wordChoice{\choice{$\R$}\choice{$\R^2$}\choice[correct]{$\R^3$}}. 
\item The tangent plane to $z=f(x,y)$ at $(3,5,f(3,5))$  is a subset of \wordChoice{\choice{$\R$}\choice{$\R^2$}\choice[correct]{$\R^3$}}.
\end{itemize}

\end{exercise}
