\begin{exercise}

Which option below is most consistent with the data available in the table of values?

\[
\begin{array}{c|c}
 x & f(x)\\ \hline
0.9 & 1.8\\ 
 1 & 2\\ 
 1.1 & 2.3\\
 1.2 & 2.7\\ 
 1.3 & 3.2\\ 
\end{array}
\]

\begin{multipleChoice}
\choice[correct]{$f'(1)>0$ and $f''(1)>0$}
\choice{$f'(1)>0$ and $f''(1)<0$}
\choice{$f'(1)<0$ and $f''(1)>0$}
\choice{$f'(1)<0$ and $f''(1)<0$}
\end{multipleChoice}

\end{exercise}
