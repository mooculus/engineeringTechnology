\begin{exercise}

For each of the functions, given by its graph, find the domain of the function and determine if the function is odd, even, or neither.


\begin{image}
  \begin{tikzpicture}
    \begin{axis}[
        xmin=-4.3,xmax=4.3,ymin=-4.3,ymax=1.3,
        clip=false,
        unit vector ratio*=1 1 1,
        axis lines=center,
        grid = major,
        ytick={-4,-3,...,1},
    xtick={-4,-3,...,4},
        xlabel=$x$, ylabel=$y$,
        every axis y label/.style={at=(current axis.above origin),anchor=south},
        every axis x label/.style={at=(current axis.right of origin),anchor=west},
      ]
      \addplot[very thick, penColor] plot coordinates {(-4,1) (-2,0)};
      \addplot[very thick, penColor, domain=-2:0] {-(x+2)^2};
      \addplot[very thick, penColor, domain=0:2] {-(x-2)^2};
      \addplot[very thick, penColor] plot coordinates {(2,0) (4,1)};
      
      \node[penColor] at (axis cs:2, -1.5) [penColor] {$y=f(x)$};       
      \end{axis}`
  \end{tikzpicture}
\end{image}

The domain of $f$ is $[\answer{-4},\answer{4}]$.
\begin{multipleChoice}
\choice{$f$ is odd.}
\choice[correct]{$f$ is even.}
\choice{$f$ is neither.}
\end{multipleChoice}
