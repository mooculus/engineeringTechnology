\begin{exercise}
  Find the domain of $f$.
  \[
  \left[\answer{0},\answer{4}\right]
  \]
  Find the range of $f$.
  \[
  \left[\answer{0},\answer{2}\right) \cup \left[\answer{3},\answer{4}\right)
  \]
\end{exercise}

\begin{exercise}
  Find an expression for $f(x)$ if $0<x<2$.
  \[
  f(x) = \answer{-x/2 + 4} \quad\text{for $0<x<2$.}
  \]
\end{exercise}


\begin{exercise}
  The graph of a function $g$ is given below. It was obtained by
  shifiting and scaling the graph of $f$.
\begin{image}
  \begin{tikzpicture}
    \begin{axis}[
        xmin=-5,xmax=5,ymin=-5,ymax=5,
        clip=false,
        unit vector ratio*=1 1 1,
        axis lines=center,
        grid = major,
        ytick={-5,-4,...,5},
	xtick={-5,-4,...,5},
        xlabel=$x$, ylabel=$y$,
        every axis y label/.style={at=(current axis.above origin),anchor=south},
        every axis x label/.style={at=(current axis.right of origin),anchor=west},
      ]
      \addplot[very thick,penColor] plot coordinates {(-2,-2) (0,-1.5)};
      \addplot[very thick,penColor,domain=0:2] {-sqrt(abs(2*(x-2)))/2};

      \addplot[color=penColor,fill=white,only marks,mark=*] coordinates{(-2,-2)};  %% open hole
      \addplot[color=penColor,fill=white,only marks,mark=*] coordinates{(0,-1)};  %% open hole

      \addplot[color=penColor,fill=penColor,only marks,mark=*] coordinates{(-2,0)};  %% closed hole
      \addplot[color=penColor,fill=penColor,only marks,mark=*] coordinates{(0,-1.5)};  %% closed hole
      \addplot[color=penColor,fill=penColor,only marks,mark=*] coordinates{(2,0)};  %% closed hole

      \node[penColor] at (axis cs:-2, 2) [penColor] {$y=g(x)$};
      \end{axis}`
  \end{tikzpicture}
\end{image}
  Find the expression for the function $g$.    \[
  g(x) = \answer{-1/2} \cdot f\left(\answer{x+2}\right)    
  \]
\end{exercise}

\begin{exercise}
  Find the domain of $g$.
  \[
  \left[\answer{-2},\answer{2}\right]
  \]
  Find the range of $g$.
  \[
  \left(\answer{-2},\answer{-3/2}\right] \cup \left(\answer{-1},\answer{0}\right]
  \]
\end{exercise}


\begin{exercise}
  Let $F$ be an \textbf{odd} function defined on $[-4,4]$ such that
  $F(x) = f(x)$ if $0\le x\le 4$:
  \begin{image}
  \begin{tikzpicture}
    \begin{axis}[
        xmin=-5,xmax=5,ymin=-5,ymax=5,
        clip=false,
        unit vector ratio*=1 1 1,
        axis lines=center,
        grid = major,
        ytick={-5,-4,...,5},
	xtick={-5,-4,...,5},
        xlabel=$x$, ylabel=$y$,
        every axis y label/.style={at=(current axis.above origin),anchor=south},
        every axis x label/.style={at=(current axis.right of origin),anchor=west},
      ]
      \addplot[very thick,penColor] plot coordinates {(0,4) (2,3)};
      \addplot[very thick,penColor] plot coordinates {(0,-4) (-2,-3)};
      \addplot[very thick,penColor,domain=2:4] {sqrt(abs(2*(x-4)))};
      \addplot[very thick,penColor,domain=-4:-2] {-sqrt(abs(2*(-x-4)))};

      \addplot[color=penColor,fill=white,only marks,mark=*] coordinates{(0,4)};  %% open hole
      \addplot[color=penColor,fill=white,only marks,mark=*] coordinates{(2,2)};  %% open hole

      \addplot[color=penColor,fill=white,only marks,mark=*] coordinates{(0,-4)};  %% open hole
      \addplot[color=penColor,fill=white,only marks,mark=*] coordinates{(-2,-2)};  %% open hole

      \addplot[color=penColor,fill=penColor,only marks,mark=*] coordinates{(0,0)};  %% closed hole
      \addplot[color=penColor,fill=penColor,only marks,mark=*] coordinates{(4,0)};  %% closed hole
      \addplot[color=penColor,fill=penColor,only marks,mark=*] coordinates{(2,3)};  %% closed hole

      \addplot[color=penColor,fill=penColor,only marks,mark=*] coordinates{(-4,0)};  %% closed hole
      \addplot[color=penColor,fill=penColor,only marks,mark=*] coordinates{(-2,-3)};  %% closed hole

      \node[penColor] at (axis cs:-2, 2) [penColor] {$y=F(x)$};
      \end{axis}`
  \end{tikzpicture}
  \end{image}
  Find an expression for $F(x)$ if $-4\le x\le 0$. Your answer should
  contain letter $f$.
  \[
  F(x) = \answer{-f(-x)} \quad \text{for $-4\le x<0$.}
  \]
\end{exercise}
