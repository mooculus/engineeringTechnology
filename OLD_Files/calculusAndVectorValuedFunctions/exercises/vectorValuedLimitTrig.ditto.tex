\author{Jim Fowler}

\outcome{Compute the limit of vector-valued functions involving trigonometric expressions.}

\begin{document}
\begin{exercise}

  Let $\vec{f}(t) = \vector{-\frac{6\sin(t) + 6}{t^3-6t-6}, 2t, -\frac{6\cos(t)\sin(t)}{2t^3-3t} }$.

  Compute
  \[
    \lim_{t \to 0} \vec{f}(t) =
    \vector{ \answer{1}, \answer{0}, \answer{2}}.
  \]
  
  \begin{hint}
    We compute the limit of a vector-valued function componentwise.
  \end{hint}
\end{exercise}
