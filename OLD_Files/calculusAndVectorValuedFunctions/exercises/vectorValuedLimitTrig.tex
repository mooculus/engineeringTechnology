\author{Jim Fowler}

\outcome{Compute the limit of vector-valued functions involving trigonometric expressions.}

\begin{document}
\begin{exercise}

  Let $\vec{f}(t) = \vector{\frac{\sin(t)}{t},\frac{\cos(t)-1}{t},\frac{t}{\sin(3t)}}$.

  Compute
  \[
    \lim_{t \to 0} \vec{f}(t) =
    \vector{ \answer{1}, \answer{0}, \answer{1/3}}.
  \]
  
  \begin{hint}
    We compute the limit of a vector-valued function componentwise.
  \end{hint}
\end{exercise}
