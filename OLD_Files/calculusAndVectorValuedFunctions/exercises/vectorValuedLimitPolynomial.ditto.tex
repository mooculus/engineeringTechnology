\author{Jim Fowler}

\outcome{Compute the limit of vector-valued functions involving polynomial expressions.}

\begin{document}
\begin{exercise}
  Let $\vec{f}(t) = \vector{t^2+2t, 2t^2+2t+2,2t+2 }$.

  Compute
  \[
    \lim_{t \to -1} \vec{f}(t) = \vector{ \answer{-1}, \answer{2}, \answer{0} }.
  \]
  
  \begin{hint}
    We compute the limit of a vector-valued function componentwise.
  \end{hint}
\end{exercise}
\end{document}
