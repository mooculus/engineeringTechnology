\outcome{Compute an antiderivative of a vector-valued function.}

\author{Jim Fowler}

\begin{document}
\begin{exercise}
  Let $\vec{f}(t) = \vector{1,2,3}$.

  Find $\vec{v}(t)$ so that $\vec{v}(0) = \vector{0,0,0}$ and $\vec{v}'(t) = \vec{f}(t)$.
  \[
    \vec{v}(t) = \vector{\answer{t},\answer{2t},\answer{3t}}.
  \]
\end{exercise}
