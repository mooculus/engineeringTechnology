\begin{exercise}

The region $R$ bounded by $x=4y^2$ and $x+4y=8$ is shown below: 

   \begin{image}

            		domain=-.2:20.2, ymax=1.8,xmax=18.2, ymin=-2.8, xmin=-.8,
            		axis lines =center, xlabel=$x$, ylabel=$y$,
            		every axis y label/.style={at=(current axis.above origin),anchor=south},

            	\addplot [draw=penColor,domain=0:9,very thick,smooth,samples=200] {sqrt(1/4*x)};

            	\addplot [draw=penColor2,very thick,smooth] {2-1/4*x};
	                            
            	\addplot [name path=A,domain=0:4,draw=none,samples=200] {sqrt(1/4*x)};   
            	\addplot [name path=B,domain=0:15,draw=none,samples=200] {-1*sqrt(1/4*x)};
	        \addplot [name path=C,domain=4:16,draw=none] {2-1/4*x};

	        \addplot [fillp] fill between[of=C and B];
                                   
            	\node at (axis cs:11,1.55) [penColor] {$x=4y^2$};
            	\node at (axis cs:15,-.9) [penColor2] {$x+4y=8$};
	    
	      \end{axis}
            \end{tikzpicture}
            \end{image}


If we want to find the area of $R$ by setting up an integral or sum of integrals with respect to $x$: 

\begin{multipleChoice}
\choice{one integral is needed to compute the area.}
\choice[correct]{two integrals are needed to compute the area.}  
\end{multipleChoice}



\begin{exercise}
In fact, $A=\int_{0}^{\answer{4}} \answer{\sqrt{\frac{x}{4}}-\left(-\sqrt{\frac{x}{4}}\right) } \d x  +\int_{\answer{4}}^{\answer{16}} \answer{\left( 2-\frac{1}{4}x \right) - \left( -\sqrt{\frac{x}{4}} \right)}  \d x$.
\end{exercise}

\begin{exercise}
By evaluating the sum of integrals above, we find that the area of the region $R$ is $\answer{18}$. 
\end{exercise}

\end{exercise}
