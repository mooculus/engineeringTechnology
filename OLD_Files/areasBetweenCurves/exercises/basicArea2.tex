\outcome{Find the bounded area between two curves}

\begin{document}
\begin{exercise}
	Set up an integral that computes the area between $y=6x+12$ and $y=6x^2$, then evaluate it.
            \begin{image}
            \begin{tikzpicture}
            	\begin{axis}[
            		domain=-2:4, ymax=32,xmax=4.5, ymin=-4, xmin=-2.75,
            		axis lines =center, xlabel=$x$, ylabel=$y$,
            		every axis y label/.style={at=(current axis.above origin),anchor=south},
            		every axis x label/.style={at=(current axis.right of origin),anchor=west},
            		axis on top,
            		]
                      
            	\addplot [draw=penColor,very thick,smooth] {6*x+12};
            	\addplot [draw=penColor2,very thick,smooth] {6*x^2};
                       
            	\addplot [name path=A,domain=-1:2,draw=none] {6*x+12};   
            	\addplot [name path=B,domain=-1:2,draw=none] {6*x^2};
            	\addplot [fillp] fill between[of=A and B];
                      
                      
            	\node at (axis cs:3.5,25) [penColor] {$y=6x+12$};
            	\node at (axis cs:-2,26) [penColor2] {$y=6x^2$};
            	\end{axis}
            \end{tikzpicture}
            \end{image}

	\[
	A= \int_{x=\answer{-1}}^{x=\answer{2}}
	\answer{6x+12-6x^2}\d x = \answer{27}
	\]
\end{exercise}
