\begin{exercise}
Set up, but DO NOT EVALUATE, an integral or sum of integrals with respect to $x$ that would compute the area of $R$. 

\begin{multipleChoice}
\choice{One integral is needed to compute the area.}
\choice[correct]{Two integrals are needed to compute the area.}  
\end{multipleChoice}

\begin{exercise}

In fact:

\[
A=\int_{-1}^{\answer{-1/\sqrt{2}}} \answer{\sqrt{1-x^2} - (-\sqrt{1-x^2})} \d x+\int_{\answer{-1/\sqrt{2} }}^{\answer{1/\sqrt{2}}} \answer{\sqrt{1-x^2} - x} \d x.
\]

\end{exercise}
\end{exercise}

\begin{exercise}
Set up, but DO NOT EVALUATE, an integral or sum of integrals with respect to $y$ that would compute the area of $R$. 

\begin{multipleChoice}
\choice{One integral is needed to compute the area.}  
\choice[correct]{Two integrals are needed to compute the area.}
\end{multipleChoice}

\begin{exercise}
In fact: 

\[
A=\int_{-1/\sqrt{2}}^{\answer{1/\sqrt{2}}} \answer{y - (-\sqrt{1-y^2})} \d y+\int_{\answer{1/\sqrt{2} }}^{\answer{1}} \answer{\sqrt{1-y^2} - (-\sqrt{1-y^2})} \d y
\]

\end{exercise}	
\end{exercise}

\end{exercise}
