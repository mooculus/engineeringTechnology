\begin{exercise}
  Suppose $\vec{v} : \R \to \R^2$ is the function given by the rule $\vec{v}(t) = \vector{t^2,t^3}$.

  Which graph below displays the parametric curve traced out by this function?
  \begin{multipleChoice}
    \choice{\begin{tikzpicture}
        \draw[->] (-1,0) -- (1,0);
        \draw[->] (0,-1) -- (0,1);
        \draw[domain=-1:1,smooth,variable=\x,penColor, ultra thick]  plot ({-\x*\x},{\x*\x*\x});
      \end{tikzpicture}}    
    \choice[correct]{\begin{tikzpicture}
        \draw[->] (-1,0) -- (1,0);
        \draw[->] (0,-1) -- (0,1);
        \draw[domain=-1:1,smooth,variable=\x,penColor, ultra thick]  plot ({\x*\x},{\x*\x*\x});
      \end{tikzpicture}}
    \choice{\begin{tikzpicture}
        \draw[->] (-1,0) -- (1,0);
        \draw[->] (0,-1) -- (0,1);
        \draw[domain=-1:1,smooth,variable=\x,penColor,ultra thick]  plot ({\x*\x*\x},{\x*\x});
      \end{tikzpicture}}
    \choice{\begin{tikzpicture}
        \draw[->] (-1,0) -- (1,0);
        \draw[->] (0,-1) -- (0,1);
        \draw[domain=-1:1,smooth,variable=\x,penColor, ultra thick]  plot ({\x*\x*\x},{-\x*\x});
    \end{tikzpicture}}    
    
  \end{multipleChoice}

\end{exercise}
