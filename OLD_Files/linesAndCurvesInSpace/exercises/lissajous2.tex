\begin{exercise}
  Suppose $\vec{v} : \R \to \R^2$ is the function given by the rule $\vec{v}(t) = \vector{\cos (2t),\sin (3t)}$.

  Which graph below displays the parametric curve traced out by this function?
  \begin{multipleChoice}
    \choice[correct]{\begin{tikzpicture}
        \draw[->] (-2,0) -- (2,0);
        \draw[->] (0,-2) -- (0,2);
        \draw[domain=0:7,smooth,samples=100,variable=\x,ultra thick, penColor]  plot ({cos(deg(2*\x))},{sin(deg(3*\x))});
      \end{tikzpicture}}
    \choice{\begin{tikzpicture}
        \draw[->] (-2,0) -- (2,0);
        \draw[->] (0,-2) -- (0,2);
        \draw[domain=0:7,smooth,samples=100,variable=\x,ultra thick, penColor]  plot ({cos(deg(3*\x))},{sin(deg(2*\x))});
      \end{tikzpicture}}
   \choice{\begin{tikzpicture}
        \draw[->] (-2,0) -- (2,0);
        \draw[->] (0,-2) -- (0,2);
        \draw[domain=0:7,smooth,samples=100,variable=\x,ultra thick, penColor]  plot ({cos(deg(\x))},{sin(deg(3*\x))});
      \end{tikzpicture}}
    \choice{\begin{tikzpicture}
        \draw[->] (-2,0) -- (2,0);
        \draw[->] (0,-2) -- (0,2);
        \draw[domain=0:7,smooth,samples=100,variable=\x,ultra thick, penColor]  plot ({cos(deg(3*\x))},{sin(deg(\x))});
      \end{tikzpicture}}        
  \end{multipleChoice}

\end{exercise}
