\begin{exercise}
  Suppose $\vec{v} : [0,\infty) \to \R^2$ is the function given by the rule $\vec{v}(t) = \vector{e^t \cos t, e^t \sin t}$.

  Which graph below displays a piece of parametric curve traced out by this function?
  \begin{multipleChoice}
    \choice[correct]{\begin{tikzpicture}
        \draw[->] (-2,0) -- (2,0);
        \draw[->] (0,-2) -- (0,2);
        \draw[domain=0:30,smooth,samples=100,variable=\x,ultra thick, penColor]  plot ({exp(\x/50)*cos(deg(\x))},{exp(\x/50)*sin(deg(\x))});
      \end{tikzpicture}}
    \choice{\begin{tikzpicture}
        \draw[->] (-2,0) -- (2,0);
        \draw[->] (0,-2) -- (0,2);
        \draw[domain=0:30,smooth,samples=100,variable=\x,ultra thick, penColor]  plot ({exp(\x/50)*cos(-deg(\x))},{exp(\x/50)*sin(-deg(\x))});
      \end{tikzpicture}}    
    \choice{\begin{tikzpicture}
        \draw[->] (-2,0) -- (2,0);
        \draw[->] (0,-2) -- (0,2);
        \draw[domain=0:30,smooth,samples=100,variable=\x,ultra thick, penColor]  plot ({exp(\x/50)*sin(deg(\x))},{exp(\x/50)*cos(deg(\x))});
      \end{tikzpicture}}    
    \choice{\begin{tikzpicture}
        \draw[->] (-2,0) -- (2,0);
        \draw[->] (0,-2) -- (0,2);
        \draw[domain=0:30,smooth,samples=100,variable=\x,ultra thick, penColor]  plot ({exp(\x/50)*sin(-deg(\x))},{exp(\x/50)*cos(-deg(\x))});
      \end{tikzpicture}}    
  \end{multipleChoice}

\end{exercise}
