\begin{exercise}
  Consider $F(x,y) = e^{x^2+y^2}$.
  \begin{itemize}
  \item The domain is a part of \wordChoice{\choice{$\R$}\choice[correct]{$\R^2$}} for which 
    \[
    \answer{-\infty} < x < \answer{\infty}\quad\text{and}\quad \answer{-\infty} < y < \answer{\infty}
    \]
  \item The range is a part of \wordChoice{\choice[correct]{$\R$}\choice{$\R^2$}} for which
    \[
     \answer{1} \leq z < \answer{\infty}
    \]    
  \end{itemize}
\end{exercise}
