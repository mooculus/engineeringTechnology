\begin{exercise}
  Consider $H(x,y) = \frac{1}{\ln(x+2y)}$
  \begin{itemize}
  \item The domain is: All ordered pairs $(x,y)$ above the line
    $x+2y=\answer{0}$ except the line $x+2y=\answer{1}$.
  \item The range is all real numbers except $\answer{0}$.
  \end{itemize}
\end{exercise}
