\begin{document}
\begin{exercise}

Find
\[
\int_{1}^{3}(3x^2-2x+1)\d x=\answer{20}
\]

\begin{hint}
Use the Second Fundamental Theorem of Calculus: 

\[
\int_{1}^{3}f(x)\d x=F(3)-F(1), 
\]

where $F$ is an antiderivative of $f$.
\end{hint}
\end{exercise}
