\outcome{Use the total differential to approximate the value of a function.}

\begin{document}
\begin{exercise}
  Use a linear approximation to approximate:
  \[
  (.97)(1.2)
  \begin{prompt}
    \approx\answer{1.17}
  \end{prompt}
  \]
  \begin{hint}
    Set $F(x,y)=xy$
  \end{hint}
  \begin{hint}
    Compute $\grad F$
  \end{hint}
  \begin{hint}
    $\d \vec{x} = \vector{-.03,.2}$
  \end{hint}
  \begin{hint}
    The linear approximation is $\grad F(1,1)\dotp \d \vec{x} +
    F(1,1)$.
  \end{hint}

\end{exercise}
