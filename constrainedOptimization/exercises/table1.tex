\begin{exercise}
  Let $F:\R^2\to\R$ be a differentiable function that is roughly
  described by the following table of values:
  \begin{image}
  \begin{tikzpicture}[x=1cm,y=.75cm]
    \draw (0,0) grid [step=1] (6,5);
    
    \draw[ultra thick] (0,1)--(6,1);
    \draw[ultra thick] (1,0)--(1,5);
    
    \draw (0,0) -- (1,1);
    \node at (.4,.9) [below left,inner sep=1pt] {\small$y$};
    \node at (0.6,.1) [above right,inner sep=1pt] {\small$x$};
    
    %% y-values
    \node at (0.5,4.5) {$4$};
    \node at (0.5,3.5) {$3$};
    \node at (0.5,2.5) {$2$};
    \node at (0.5,1.5) {$1$};

    
    %% z-values
    %% top
    \node at (1.5,4.5) {$0$};
    \node at (2.5,4.5) {$-3$};
    \node at (3.5,4.5) {$-8$};
    \node at (4.5,4.5) {$-10$};
    \node at (5.5,4.5) {$-11$};
    
    %% 
    \node at (1.5,3.5) {$1$};
    \node at (2.5,3.5) {$-2$};
    \node at (3.5,3.5) {$-7$};
    \node at (4.5,3.5) {$-12$};
    \node at (5.5,3.5) {$-13$};
    
    %% 
    \node at (1.5,2.5) {$4$};
    \node at (2.5,2.5) {$-1$};
    \node at (3.5,2.5) {$-3$};
    \node at (4.5,2.5) {$-5$};
    \node at (5.5,2.5) {$-9$};
    
    %% 
    \node at (1.5,1.5) {$6$};
    \node at (2.5,1.5) {$3$};
    \node at (3.5,1.5) {$-1$};
    \node at (4.5,1.5) {$-2$};
    \node at (5.5,1.5) {$-4$};
    
    %% bottom row
    \node at (1.5,.5) {$-2$};
    \node at (2.5,.5) {$-1$};
    \node at (3.5,.5) {$0$};
    \node at (4.5,.5) {$1$};
    \node at (5.5,.5) {$2$};
  \end{tikzpicture}
  \end{image}
  Consider the closed set defined by:
  \[
  \vecl(t) = \vector{-1,1}+t\vector{3,3}
  \]
  as $t$ runs from $0$ to $1$. Find the absolute extrema on this set.


  \begin{prompt}
     The absolute maximum of $F$ when constrained to the closed set is:
    \[
    F\left(\answer{-1},\answer{1}\right) = \answer{3}
    \]
    The absolute minimum of $F$ when constrained to the closed set is:
    \[
    F\left(\answer{1},\answer{3}\right) = \answer{-12}
    \]
  \end{prompt}
\end{exercise}
