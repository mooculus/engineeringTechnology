\outcome{Use the total differential to approximate the value of a function.}

\begin{document}
\begin{exercise}
  Use a linear approximation to approximate:
  \[
  \sin(.3)\cdot \cos(-.3)
  \begin{prompt}
    \approx\answer{0.3}
  \end{prompt}
  \]
  \begin{hint}
    Set $F(x,y)=\sin(x)\cos(y)$.
  \end{hint}
  \begin{hint}
    Compute $\grad F$
  \end{hint}
  \begin{hint}
    $\d \vec{x} = \vector{.3,-.3}$
  \end{hint}
  \begin{hint}
    The linear approximation is $\grad F(0,0)\dotp \d \vec{x} +
    F(0,0)$.
  \end{hint}

\end{exercise}
