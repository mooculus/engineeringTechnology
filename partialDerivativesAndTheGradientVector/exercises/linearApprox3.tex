\begin{document}
\begin{exercise}
  Use a linear approximation to approximate:
  \[
  .99\cdot \sqrt{16.3}e^{-0.02}
  \begin{prompt}
    \approx\answer{3.9175}
  \end{prompt}
  \]
  \begin{hint}
    Set $F(x,y,z)=x\sqrt{y}e^z$.
  \end{hint}
  \begin{hint}
    Compute $\grad F$
  \end{hint}
  \begin{hint}
    $\d \vec{x} = \vector{-.01,.3,-.02}$
  \end{hint}
  \begin{hint}
    The linear approximation is $\grad F(1,16,0)\dotp \d \vec{x} +
    F(1,16,0)$.
  \end{hint}

\end{exercise}
