\begin{exercise}
Solve the following equation for x.
\[
\ln{(x^2+x)}-\ln{x}=4
\]
\begin{hint}

Use the properties of logarithms.
\[
\ln{(x^2+x)}-\ln{x}=\ln{\frac{x^2+x}{x}}
\]
\end{hint}
\begin{hint}
Simplify. 
\[
\ln{\frac{x^2+x}{x}}=4
\]
\end{hint}
\begin{hint}
Simplify. Remember,
\[
\ln{(x+1)}=4 
\]
is equivalent to
\[
x+1=e^4
\]
\end{hint}
ANSWER: 
\[
x=\answer{e^4-1}
\]
\end{exercise}
