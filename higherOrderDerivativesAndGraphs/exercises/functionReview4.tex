\begin{exercise}

The (entire) graph of a function $f$ is given below.

\begin{image}
  \begin{tikzpicture}
    \begin{axis}[
        xmin=-8.8,xmax=8.8,ymin=-8.8,ymax=6.8,
        clip=true,
        unit vector ratio*=1 1 1,
        axis lines=center,
        grid = major,
        ytick={-8,-6,...,6},
    xtick={-8,-6,...,8},
        xlabel=$x$, ylabel=$y$,
        every axis y label/.style={at=(current axis.above origin),anchor=south},
        every axis x label/.style={at=(current axis.right of origin),anchor=west},
      ]
      \draw[very thick,penColor] (axis cs:-8,0) -- (axis cs:0,-8);
      \addplot[very thick,penColor,domain=0:4,samples=50] plot{-x^2/4 + 6};
      \addplot[very thick,penColor,domain=4:8,samples=50] plot{sqrt(8-x)};
      
      \addplot[penColor,only marks,mark=*] coordinates{(0,2) (8,0)};
      \addplot[penColor,fill=white,only  marks,mark=*] coordinates{(-8,0) (0,-8) (0,6) (4,2)};
      
      \node at (axis cs:3,-3) [penColor] {$y=f(x)$};
      \end{axis}`
  \end{tikzpicture}
\end{image}

The domain of $f$ (from left to right) is
\[
\left(\answer{-8},\answer{4}\right)\cup\left(\answer{4},\answer{8}\right].
\]

The range of $f$ is
\[
\left(\answer{-8},\answer{6}\right).
\]

\begin{align*}
f(0) &= \answer{2}\\
f(4) &= \answer{DNE}\\
f(8) &= \answer{0}\\
f^{-1}(-8) &= \answer{DNE}\\
f(-2) &= \answer{-6}\\
f(-6) &= \answer{-2}\\
f^{-1}(0) &= \answer{8}\\
f^{-1}(2) &= \answer{0}\\
f^{-1}(6) &= \answer{DNE}\\
f^{-1}(-6) &= \answer{-2}
\end{align*}

\end{exercise}
