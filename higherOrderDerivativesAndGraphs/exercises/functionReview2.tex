\begin{exercise}

The (entire) graph of a function $f$ is given below.

\begin{image}
  \begin{tikzpicture}
    \begin{axis}[
        xmin=-0.8,xmax=5.8,ymin=-3.8,ymax=7.8,
        clip=true,
        unit vector ratio*=1 1 1,
        axis lines=center,
        grid = major,
        ytick={-3,-2,...,7},
    xtick={-0,1,...,5},
        xlabel=$x$, ylabel=$y$,
        every axis y label/.style={at=(current axis.above origin),anchor=south},
        every axis x label/.style={at=(current axis.right of origin),anchor=west},
      ]
      \addplot[very thick,penColor,domain=0:2,samples=50] plot{6-x^2};
      \draw[very thick,penColor] (axis cs:2,0) -- (axis cs:5,-3);
      
      \addplot[penColor,only marks,mark=*] coordinates{(0,6) (2,7) (5,-3)};
      \addplot[penColor,fill=white,only marks,mark=*] coordinates{(2,2) (2,0)};
      
      \node at (axis cs:4,4.5) [penColor] {$y=f(x)$};
      \end{axis}`
  \end{tikzpicture}
\end{image}

The domain of $f$ is
\[
\left[\answer{0},\answer{5}\right].
\]

The range of $f$ (from bottom to top) is
\[
\left[\answer{-3},\answer{0}\right)\cup\left(\answer{2},\answer{6}\right]\cup\left\{\answer{7}\right\}.
\]

$f$ is one-to-one.
\begin{multipleChoice}
\choice[correct]{True}
\choice{False}
\end{multipleChoice}

The domain of $f^{-1}$ (from left to right) is
\[
\left[\answer{-3},\answer{0}\right)\cup\left(\answer{2},\answer{6}\right]\cup\left\{\answer{7}\right\}.
\]

The range of $f^{-1}$ is
\[
\left[\answer{0},\answer{5}\right].
\]

\begin{align*}
f(5) &= \answer{-3}\\
f(2) &= \answer{7}\\
f(0) &= \answer{6}\\
f^{-1}(7) &= \answer{2}\\
f^{-1}(6) &= \answer{0}\\
f^{-1}(-3) &= \answer{5}
\end{align*}

\end{exercise}
