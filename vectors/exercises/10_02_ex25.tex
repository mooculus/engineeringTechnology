\begin{exercise}
Find the unit vector $\uvec{u}$ in the first quadrant of $\R^2$ that
makes an angle of $5\pi/18$ radians with the $x$-axis.
\begin{prompt}
\[
\uvec{u}=\vector{\answer{\cos\left(\frac{5\pi}{18}\right)},\answer{\sin\left(\frac{5\pi}{18}\right)}}
\]
\end{prompt}

\end{exercise}
