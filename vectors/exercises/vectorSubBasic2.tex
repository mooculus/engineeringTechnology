\begin{exercise}
  Consider the vector $\vec{v}=\vector{-4,3}$ and vector $\vec{w}$ below:
  \begin{image}
    \begin{tikzpicture}
      \begin{axis}[
          xmin=-3,xmax=4,ymin=-1,ymax=3,
          clip=false,
          axis lines=center,
          unit vector ratio*=1 1 1,
          xlabel=$x$, ylabel=$y$,
          grid = major,
          every axis y label/.style={at=(current axis.above origin),anchor=south},
          every axis x label/.style={at=(current axis.right of origin),anchor=west},
        ]
          \addplot[very thick,penColor,->] plot coordinates {(-2,2) (3,0)};
      \end{axis}
    \end{tikzpicture}
  \end{image}
  Compute $\vec{v}-\vec{w}$.
  \begin{prompt}
    \[
    \vec{v} - \vec{w} = \vector{\answer{-9},\answer{5}}
    \]
  \end{prompt}
\end{exercise}
