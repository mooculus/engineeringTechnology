\begin{exercise}
Let $\vec{u}=\vector{1,-2}$ and $\vec{v}=\vector{1,1}$ be two
vectors. Find the vector $\vec{x}$ such that
$\vec{u}+\vec{x}=2\vec{v}-\vec{x}$.
\begin{prompt}
\[
\vec{x}=\vector{\answer{1/2},\answer{2}}
\]
\end{prompt}

\end{exercise}
