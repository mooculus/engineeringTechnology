\outcome{Add and subtract vectors.}

\begin{document}
\begin{exercise}
  Consider the vectors below:
  \begin{image}
    \begin{tikzpicture}
        \begin{axis}[
            unit vector ratio*=1 1 1,
            axis lines=none
          ]
          \addplot[very thick,penColor,->] plot coordinates {(0,0) (-2,-3)};
          \addplot[very thick,penColor2,->] plot coordinates {(0,0) (2,-1)};
          \addplot[very thick,penColor3,->] plot coordinates {(0,0) (0,-4)};
          \addplot[very thick,penColor4,->] plot coordinates {(0,0) (1.03,-3.87)};
          \addplot[very thick,penColor5,->] plot coordinates {(0,0) (1.79,-3.58)};
          \node[left] at (axis cs:-1, -1.5) [penColor] {$\vec{v}$};
          \node[above] at (axis cs:1, -.5) [penColor2] {$\vec{w}$};
          \node[left] at (axis cs:0, -2) [penColor3] {$\vec{a}$};
          \node[above left] at (axis cs:.5,-1.9) [penColor4] {$\vec{b}$};
          \node[above] at (axis cs:1, -1.8) [penColor5] {$\vec{c}$};
        \end{axis}
    \end{tikzpicture}
  \end{image}
  Which of the following is equal to $\vec{v}+\vec{w}$?
  \begin{multipleChoice}
    \choice[correct]{Vector $\vec{a}$.}
    \choice{Vector $\vec{b}$.}
    \choice{Vector $\vec{c}$.}
    \choice{None of these.}
  \end{multipleChoice}
\end{exercise}
