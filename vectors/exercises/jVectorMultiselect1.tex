\begin{exercise}
Select all of the following statements that must be true:

\begin{selectAll}
\choice[correct]{Given any nonzero vector $\vec{u}$, the vector $\frac{\vec{u}}{|\vec{u}|}$ has magnitude $1$.}
\choice{Given any nonzero vectors $\vec{u}$ and $\vec{v}$ with the same dimension, $|\vec{u}+\vec{v}| = |\vec{u}| + |\vec{v}|$.}
\choice[correct]{If $\vec{u} = 3\vec{v}$, then  $|\vec{u}+\vec{v}| = |\vec{u}| + |\vec{v}|$}
\choice[correct]{If $|\vec{u}| = 5$, then $|6\vec{u}| = 30$.}
\choice[correct]{If $\vec{u} = \vector{u_1,u_2,u_3}$ and $\vec{v} = \vector{v_1,v_2,v_3}$, then $\vec{u} = \vec{v}$ if and only if:

\[
u_1=v_1 \qquad u_2=v_2 \qquad u_3=v_3 \qquad 
\]}
\choice{If $|\vec{u}| = |\vec{v}|$, then $\vec{u} = \vec{v}$.}
\choice[correct]{If $\vec{u} = \vec{v}$, then $|\vec{u}| = |\vec{v}|$.}
\end{selectAll}

\end{exercise}
