\begin{exercise}
  
Let $\vec{u}=\left\langle1,-2\right\rangle$ and
$\vec{v}=\left\langle1,1\right\rangle$ be two vectors. Find the
vector $2\vec{u}-3\vec{v}$.
\begin{prompt}
\[
2\vec{u}-3\vec{v}=\vector{\answer{-1},\answer{-7}}
\]
\end{prompt}

\end{exercise}
