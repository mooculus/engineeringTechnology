\outcome{Identify parallel vectors.}

\begin{document}
\begin{exercise}
  True or false: Two vectors $\vec{v}$ and $\vec{w}$ are parallel if 
  \[
  \vec{v} = s\cdot \vec{w}\
  \]
  for some scalar $s$.
  \begin{multipleChoice}
    \choice{true}
    \choice[correct]{false}
  \end{multipleChoice}
\end{exercise}
