\begin{exercise}
  Consider the vectors below:
  \begin{image}
    \begin{tikzpicture}
        \begin{axis}[
            unit vector ratio*=1 1 1,
            axis lines=none
          ]
          \addplot[very thick,penColor,->] plot coordinates {(0,0) (0,2)};
          \addplot[very thick,penColor2,->] plot coordinates {(0,0) (4,1)};
          \addplot[very thick,penColor3,->] plot coordinates {(0,0) (1.21,4.85)};
          \addplot[very thick,penColor4,->] plot coordinates {(0,0) (3.08,3.94)};
          \addplot[very thick,penColor5,->] plot coordinates {(0,0) (4,3)};
          \node[left] at (axis cs:0, 1) [penColor] {$\vec{v}$};
          \node[above] at (axis cs:2, .5) [penColor2] {$\vec{w}$};
          \node[left] at (axis cs:.605, 2.4) [penColor3] {$\vec{a}$};
          \node[above left] at (axis cs:1.5, 2) [penColor4] {$\vec{b}$};
          \node[above] at (axis cs:2, 1.5) [penColor5] {$\vec{c}$};
        \end{axis}
    \end{tikzpicture}
  \end{image}
  Which of the following is equal to $\vec{v}+\vec{w}$?
  \begin{multipleChoice}
    \choice{Vector $\vec{a}$.}
    \choice{Vector $\vec{b}$.}
    \choice[correct]{Vector $\vec{c}$.}
    \choice{None of these.}
  \end{multipleChoice}
\end{exercise}
