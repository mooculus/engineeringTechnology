\begin{exercise}
How many integrals are needed to express the volume of the solid of revolution with respect to $x$? 
\begin{multipleChoice}
\choice{One integral is needed.}
\choice[correct]{Two integrals are needed.}
\choice{More than two integrals are needed.}
\end{multipleChoice}

Which method should be used to find the volume?
\begin{multipleChoice}
\choice{Shell Method}
\choice[correct]{Washer Method}
\end{multipleChoice}


In fact: 
\[
V=\int_{x={\frac{1}{2}}}^{x=\answer{1}} \answer{\pi \left((22x-8)^2-(2x+2)^2\right)} \d x 
	+\int_{x=\answer{1}}^{x=\answer{3}} \answer{\pi \left(\left(\frac{9}{x} +5 \right)^2-(2x+2)^2\right)} \d x 
\]

\end{exercise}
\end{exercise}
