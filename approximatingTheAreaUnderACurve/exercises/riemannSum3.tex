\begin{exercise}

Consider the function
\[
f(x) = x+x^2
\]
on the interval $[0,4]$.  We will approximate the area of the region bounded by the graph of $f$ and the $x$-axis.  See the figure below.

\begin{image}
  \begin{tikzpicture}
    \begin{axis}[
        xmin=-0.3,xmax=4.3,ymin=-0.3,ymax=20.3,
        clip=true,
        unit vector ratio*=6 1 1,
        axis lines=center,
        grid = major,
        ytick={0,4,...,20},
        xtick={0,0.5,...,4},
        xlabel=$x$, ylabel=$y$,
        every axis y label/.style={at=(current axis.above origin),anchor=south},
        every axis x label/.style={at=(current axis.right of origin),anchor=west},
      ]      
      \pgfplotsinvokeforeach{0.5,1,...,4}{\draw[very thick,penColor2,fill,fill opacity=0.3] (axis cs:{#1-0.5},0) rectangle (axis cs:{#1},{#1+#1^2});}
      
      \addplot[very thick,penColor,domain=0:4] plot{x+x^2};
      
      \node at (axis cs:2,15) {$y=f(x)$};
      \end{axis}`
  \end{tikzpicture}
\end{image}

The image depicts a \wordChoice{\choice{Left}\choice[correct]{Right}\choice{Midpoint}} Riemann sum with $n=\answer{8}$ rectangles.

\begin{align*}
x_6^* &= \answer{3}\\
f(x_6^*) &= \answer{12}\\
\Delta x &= \answer{\frac{1}{2}}\\
\sum_{k=1}^n f(x_k^*)\Delta x &= \answer{\frac{69}{2}}
\end{align*}

\end{exercise}
\begin{exercise}

Consider the function
\[
f(x) = x+x^2
\]
on the interval $[0,4]$.  We will approximate the area of the region bounded by the graph of $f$ and the $x$-axis.  See the figure below.

\begin{image}
  \begin{tikzpicture}
    \begin{axis}[
        xmin=-0.3,xmax=4.3,ymin=-0.3,ymax=20.3,
        clip=true,
        unit vector ratio*=6 1 1,
        axis lines=center,
        grid = major,
        ytick={0,4,...,20},
        xtick={0,0.5,...,4},
        xlabel=$x$, ylabel=$y$,
        every axis y label/.style={at=(current axis.above origin),anchor=south},
        every axis x label/.style={at=(current axis.right of origin),anchor=west},
      ]      
      \pgfplotsinvokeforeach{0.5,1,...,4}{\draw[very thick,penColor2,fill,fill opacity=0.3] (axis cs:{#1-0.5},0) rectangle (axis cs:{#1},{#1-0.5+(#1-0.5)^2});}
      
      \addplot[very thick,penColor,domain=0:4] plot{x+x^2};
      
      \node at (axis cs:2,15) {$y=f(x)$};
      \end{axis}`
  \end{tikzpicture}
\end{image}

The image depicts a \wordChoice{\choice[correct]{Left}\choice{Right}\choice{Midpoint}} Riemann sum with $n=\answer{8}$ subintervals.

\begin{align*}
x_6^* &= \answer{2.5}\\
f(x_6^*) &= \answer{8.75}\\
\Delta x &= \answer{\frac{1}{2}}\\
\sum_{k=1}^n f(x_k^*)\Delta x &= \answer{\frac{49}{2}}
\end{align*}

\end{exercise}
