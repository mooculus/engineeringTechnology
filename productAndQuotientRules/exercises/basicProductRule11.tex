\begin{exercise}
Let $Y(n) = -5 \sin (n) \left(\sqrt{n}+3 \sin (n)\right)$. Compute and simplify:
\[
Y'(n)
\begin{prompt}=\left(\answer{-5/(2\sqrt{n})}\right)\sin{(n)} +\left(\answer{-5\sqrt{n}}\right) \cos (n)+ \left(\answer{-30}\right)\sin{(n)} \cos (n)\end{prompt}
\]
\end{exercise}
