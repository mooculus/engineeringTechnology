\begin{exercise}
  Consider $\vec{g}(t) = \vector{te^t,\ln(1+t^2),6e^{2t}}$
  \[
  \vec{g}'(t) = \vector{\answer{e^t+te^t},\answer{\frac{2t}{1+t^2}},\answer{12e^{2t}}}
  \]
\end{exercise}
