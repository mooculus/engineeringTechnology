Let
\[
A(x) = \int_0^x g(t) \d t
\]
for $0\leq x\leq4$.

The we know that $A(2)$ is \wordChoice{\choice{positive}\choice[correct]{negative}\choice{zero}}. Also $A(3.8)$ is \wordChoice{\choice[correct]{positive}\choice{negative}\choice{zero}} and $A'(3.8)$ is \wordChoice{\choice[correct]{positive}\choice{negative}\choice{zero}}.
\begin{hint}
Recall: $A(2)=\int_0^2 g(t) \d t$. This gives signed area. 
\end{hint}
\begin{hint}
Recall: $A(3.8)=\int_0^{3.8} g(t) \d t$. This gives signed area.\\

 $A(3.8)=\int_0^{3.8} g(t) \d t=\int_0^2 g(t) \d t+\int_2^{3.8} g(t) \d t$. \\
 The first integral is negative and the second positive. 
 \end{hint}
 \begin{hint}
Recall: $A'(x)=g(x)$
\end{hint}
 \begin{hint}
Recall: $A'(3.8)=g(3.8)>0$
\end{hint}
Find the solution to the initial value problem $y'(x) = g(x)$ and $y(0) = 2$.
\begin{multipleChoice}
\choice{$y(x)=g(x)$}
\choice{$y(x)=g(x)+2$}
\choice{$y(x)=A(x)$}
\choice[correct]{$y(x)=A(x)+2$}
\choice{$y(x)=g'(x)$}
\choice{$y(x)=g'(x)+2$}
\choice{None of the above}
\end{multipleChoice}

Find an expression for $\int_0^4 \left|g(t)\right| \d t$.
\begin{multipleChoice}
\choice{$A(4)$}
\choice{$A(2)-A(4)$}
\choice{$A(4)-A(2)$}
\choice[correct]{$A(4)-2A(2)$}
\choice{None of the above}
\end{multipleChoice}

Find the midpoint Riemann sum for $g$ on $[0,4]$ with 2 subintervals.
\begin{multipleChoice}
\choice{$g(1)+g(3)$}
\choice{$g(0)+g(2)$}
\choice{$g(2)+g(4)$}
\choice[correct]{$2g(1)+2g(3)$}
\choice{$2g(0)+2g(2)$}
\choice{$2g(2)+2g(4)$}
\choice{None of the above}
\end{multipleChoice}
