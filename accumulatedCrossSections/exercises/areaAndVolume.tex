\begin{exercise}

The region $R$ is bounded by $x^4+y^2=5$ and $y=2x^2$.



\begin{image}
\begin{tikzpicture}
\begin{axis}[
	domain=-6.4:6.4, ymax=2.9,xmax=1.9, ymin=-.4, xmin=-1.9,
	axis lines =center, xlabel=$x$, ylabel=$y$,
	every axis y label/.style={at=(current axis.above origin),anchor=south},

\addplot [draw=penColor,very thick,smooth,samples=200] {sqrt(5-x^4)};
\addplot [draw=penColor2,very thick,smooth] {2*x^2};

                       
\addplot [name path=A,domain=-1:1,samples=200,draw=none] {sqrt(5-x^4)};   
\addplot [name path=B,domain=-1:1,draw=none] {2*x^2};
\addplot [fillp] fill between[of=A and B];
	
		                      
\node at (axis cs:.9,.3) [penColor2] {$y=2x^2$ };
\node at (axis cs:-.15,2.5) [penColor] {$x^4+y^2=5$ };
\end{axis}
\end{tikzpicture}
\end{image}

\begin{exercise}
An integral with respect to $x$ that expresses the area of $R$ is:

\[
A=\int_{x=\answer{-1}}^{x=\answer{1}} \answer{\sqrt{5-x^4}-2x^2} \d x
\]
\end{exercise}

\begin{exercise}
The base of a certain solid is the region $R$.  Cross sections through the solid taken perpendicular to the $x$-axis are squares. 

An integral that expresses the volume of the solid is:

\[
V=\int_{x=\answer{-1}}^{x=\answer{1}} \answer{(\sqrt{5-x^4}-2x^2)^2} \d x
\]
\end{exercise}


\end{exercise}
