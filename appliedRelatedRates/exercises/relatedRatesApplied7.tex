\begin{exercise}


A bowl with a circular (top) base and cross section bounded by $y=x^3$, for $x\ge0$, and by $y=-x^3$, for $x<0$, is empty at time $t=0$ seconds.
\begin{image}
    \begin{tikzpicture}[
        declare function = {f(\x) = x^3;} ]
      \begin{axis}[
          xmin =-3.2,xmax=3.2,ymax=9.5,ymin=-.2,
          width=6in,
          height=3in,
          axis lines=center, xlabel=$x$, ylabel=$y$,
          every axis y label/.style={at=(current axis.above origin),anchor=south},
          everqy axis x label/.style={at=(current axis.right of origin),anchor=west},
          axis on top,
        ]
        \addplot [draw=none,fill=fillp!50!white,domain=-1.5:1.5, smooth] {(.666)*sqrt(1.5^2-x^2)+3.375} \closedcycle;
        \addplot [draw=none,fill=fillp,domain=-1.5:1.5, smooth] {-(.666)*sqrt(1.5^2-x^2)+3.375} \closedcycle; 
        \addplot [draw=none,fill=white,domain=0:1.6, smooth] {f(x)} \closedcycle;
        \addplot [draw=none,fill=white,domain=-1.6:0, smooth] {-f(x)} \closedcycle;
        \addplot [ultra thick,penColor, smooth,domain=2:0] {f(x)};
        \addplot [ultra thick,penColor, smooth,domain=-2:0] {-f(x)};
        \draw[penColor,very thick] (axis cs:0,8) ellipse (200 and 100);
        \draw[penColor,very thick] (axis cs:0,3.375) ellipse (150 and 100);
        \node[black] at (axis cs:2,3) {$y=x^3$};       
        \end{axis}
\end{tikzpicture}
\end{image}
If water is being added to the bowl so that the height of the water is
rising at $1/2$ cm per second, what rate is the radius of the circle,
determined by the surface of the water, changing when the height of the water is $\frac{27}{8}$ cm?

\begin{prompt}
	The radius is changing at $\answer{\frac{2}{27}}$ cm per second.
\end{prompt}

\end{exercise}
