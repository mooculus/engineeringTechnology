\begin{exercise}
  Find the critical points of $F(x,y) = x^3/3 - x + y^3/3 - 4 y$. Write
  them in the order of the quadrants which they belong.
  \begin{align*}
  \vec{c}_{I} &= \vector{\answer{1},\answer{2}}\\
  \vec{c}_{II} &= \vector{\answer{-1},\answer{2}}\\
  \vec{c}_{III} &= \vector{\answer{-1},\answer{-2}}\\
  \vec{c}_{IV} &= \vector{\answer{1},\answer{-2}}
  \end{align*}
  \begin{exercise}
  \begin{align*}
    D(\vec{c}_{I}) &= \answer{8}\\
    D(\vec{c}_{II}) &= \answer{-8}\\
    D(\vec{c}_{III}) &= \answer{8}\\
    D(\vec{c}_{IV}) &= \answer{-8}
  \end{align*}
  \begin{exercise}
    At $\vec{c}_{I}$, $F$ locally looks like:
    \begin{multipleChoice}
      \choice[correct]{an elliptic paraboloid}
      \choice{a hyperbolic paraboloid}
      \choice{neither of these}
    \end{multipleChoice}
    At $\vec{c}_{II}$, $F$ locally looks like:
    \begin{multipleChoice}
      \choice{an elliptic paraboloid}
      \choice[correct]{a hyperbolic paraboloid}
      \choice{neither of these}
    \end{multipleChoice}
        At $\vec{c}_{III}$, $F$ locally looks like:
    \begin{multipleChoice}
      \choice[correct]{an elliptic paraboloid}
      \choice{a hyperbolic paraboloid}
      \choice{neither of these}
    \end{multipleChoice}
    At $\vec{c}_{IV}$, $F$ locally looks like:
    \begin{multipleChoice}
      \choice{an elliptic paraboloid}
      \choice[correct]{a hyperbolic paraboloid}
      \choice{neither of these}
    \end{multipleChoice}
    \begin{exercise}
      The point $\vec{c}_{I}$ is:
      \begin{multipleChoice}
        \choice[correct]{a local minimum for $F$}
        \choice{a local maximum for $F$}
        \choice{neither a local minimum nor a local maximum for $F$}
      \end{multipleChoice}
      The point $\vec{c}_{II}$ is:
      \begin{multipleChoice}
        \choice{a local minimum for $F$}
        \choice{a local maximum for $F$}
        \choice[correct]{neither a local minimum nor a local maximum for $F$}
      \end{multipleChoice}
      The point $\vec{c}_{III}$ is:
      \begin{multipleChoice}
        \choice{a local minimum for $F$}
        \choice[correct]{a local maximum for $F$}
        \choice{neither a local minimum nor a local maximum for $F$}
      \end{multipleChoice}
      The point $\vec{c}_{IV}$ is:
      \begin{multipleChoice}
        \choice{a local minimum for $F$}
        \choice{a local maximum for $F$}
        \choice[correct]{neither a local minimum nor a local maximum for $F$}
      \end{multipleChoice}
    \end{exercise}
  \end{exercise}
  \end{exercise}
