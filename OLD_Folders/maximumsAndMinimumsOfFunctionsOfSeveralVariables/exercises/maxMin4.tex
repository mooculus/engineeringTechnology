\begin{exercise}
  Find the critical points of $F(x,y) = x^2 + y^3 - 3 y + 1$. Write
  them in the order of increasing $y$-values.
  \begin{align*}
    \vec{c}_1 &= \vector{\answer{0},\answer{-1}}\\
    \vec{c}_2 &= \vector{\answer{0},\answer{1}}
  \end{align*}
  \begin{exercise}
  \begin{align*}
    D(\vec{c}_1) &= \answer{-12}\\
    D(\vec{c}_2) &= \answer{12}
  \end{align*}
  \begin{exercise}
    At $\vec{c}_1$, $F$ locally looks like:
    \begin{multipleChoice}
      \choice{an elliptic paraboloid}
      \choice[correct]{a hyperbolic paraboloid}
      \choice{neither of these}
    \end{multipleChoice}
    At $\vec{c}_2$, $F$ locally looks like:
    \begin{multipleChoice}
      \choice[correct]{an elliptic paraboloid}
      \choice{a hyperbolic paraboloid}
      \choice{neither of these}
    \end{multipleChoice}
    \begin{exercise}
      The point $\vec{c}_1$ is:
      \begin{multipleChoice}
        \choice{a local minimum for $F$}
        \choice{a local maximum for $F$}
        \choice[correct]{neither a local minimum nor a local maximum for $F$}
      \end{multipleChoice}
      The point $\vec{c}_2$ is:
      \begin{multipleChoice}
        \choice[correct]{a local minimum for $F$}
        \choice{a local maximum for $F$}
        \choice{neither a local minimum nor a local maximum for $F$}
      \end{multipleChoice}
    \end{exercise}
  \end{exercise}
  \end{exercise}
\end{exercise}
