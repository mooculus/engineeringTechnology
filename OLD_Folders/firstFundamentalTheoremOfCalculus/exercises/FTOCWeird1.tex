\begin{exercise}

Is there any function $f$ satisfying
\[
\int_1^x f(t) \d t = x^2-1?
\]
If so, what function?

\begin{prompt}
  If there is such a function, write it below.  Otherwise, just write: DNE
  \begin{hint}
The function  $F(x)=x^2-1$ is an antiderivative of $f$.
  Therefore, $F'(x)=f(x)$!
    \end{hint}
  \[
  f(x) = \answer{2x}
  \]
\end{prompt}

\end{exercise}
