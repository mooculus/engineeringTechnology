\begin{exercise}

\[
\dd{x}\left[\int_0^{\sin x} \ln\left(t^3+1\right) \d t\right].
\]

\begin{multipleChoice}
\choice{$\frac{\cos x}{\sin^3(x)+1}$}
\choice{$\ln\left(\sin^3(x)+1\right)$}
\choice[correct]{$\cos(x)\ln\left(\sin^3(x)+1\right)$}
\choice{$3x^2\ln\left(\sin^3(x)+1\right)$}
\choice{$\frac{3x^2\cos(x)}{\sin^3(x)+1}$}
\choice{None of the above}
\end{multipleChoice}
