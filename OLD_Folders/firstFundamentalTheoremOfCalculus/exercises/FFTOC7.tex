\begin{exercise}

Evaluate $\dd{x}\left[\int_0^{\sin x} \ln\left(t^2+7\right) \d t\right]$.
\begin{multipleChoice}
\choice{$\frac{\cos x}{\sin^2x + 7}$}
\choice{$\ln\left(\sin^2x+7\right)$}
\choice{$2x\ln\left(\sin^2x+7\right)$}
\choice[correct]{$\cos(x)\ln\left(\sin^2x+7\right)$}
\choice{$\frac{2x\cos x}{\sin^2x + 7}$}
\choice{None of the above}
\end{multipleChoice}

Evaluate $\dd{x}\left[\int_0^4 \ln(t+7) \d t\right]$.
\begin{multipleChoice}
\choice{$\int_0^4 \frac{1}{t+7} \d t$}
\choice[correct]{$0$}
\choice{$\frac{1}{11} - \frac{1}{7}$}
\choice{$\ln(11)$}
\choice{$\ln(11)-\ln(7)$}
\choice{None of the above}
\end{multipleChoice}

\end{exercise}
