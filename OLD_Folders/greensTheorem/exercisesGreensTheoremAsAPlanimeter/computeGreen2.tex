  Let $P$ be the filled-in pentagon:
  \begin{image}
    \begin{tikzpicture}
      \begin{axis}%

          ymin=0,ymax=5,
          xlabel=$x$,ylabel=$y$,
          axis lines=center,
          every axis y label/.style={at=(current axis.above origin),anchor=south},
          every axis x label/.style={at=(current axis.right of origin),anchor=west},
          clip=false,
	  grid =major,
          xtick={0,1,...,7},
          ytick={0,1,...,5},
	]
        \addplot[line join =bevel,penColor,ultra thick] coordinates{
          (1,3)   (2,1)   (4,3)  (6,2)  (3,4) (1,3) (2,1)};
      \end{axis}
    \end{tikzpicture}
  \end{image}
  Let $\partial P$ be the boundary of $P$ drawn in a counterclockwise
  fashion. Compute:
  \[
  \oint_{\partial P} \left(\frac{x^3}{\sin(x^2)}\d x + (y^2+x)\d y\right)
  \begin{prompt}
    = \answer{5}
  \end{prompt}
  \]
\end{exercise}
