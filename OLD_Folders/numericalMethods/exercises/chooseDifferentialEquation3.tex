\begin{exercise}

Consider the slope field below:

\begin{image}
{\def\length{sqrt(1+(y*cos(1/x))^2)}
\begin{tikzpicture}
  \begin{axis}[
      xmin=-3, xmax=3,ymin=-3,ymax=3,domain=-3:3,view={0}{90},
      axis lines =center, xlabel=$x$, ylabel=$y$,
      every axis y label/.style={at=(current axis.above origin),anchor=south},
      every axis x label/.style={at=(current axis.right of origin),anchor=west},
      axis on top,
    ] 
    \addplot3 [penColor, quiver={u={1/\length}, v={(y*cos(1/x))/(\length)},scale arrows=.2},samples=20] {0};
]  \end{axis}
\end{tikzpicture}}
\end{image}


\begin{multipleChoice} 
\choice{$\frac{dy}{dx} =x \cos\left(\frac{1}{y}\right)$}
\choice[correct]{$\frac{dy}{dx} =y \cos\left(\frac{1}{x}\right)$}
\choice{$\frac{dy}{dx} =xy \cos\left(\frac{1}{x}\right)$}
\choice{$\frac{dy}{dx} =-y \cos\left(\frac{1}{x}\right)$}
\end{multipleChoice}

\end{exercise}
