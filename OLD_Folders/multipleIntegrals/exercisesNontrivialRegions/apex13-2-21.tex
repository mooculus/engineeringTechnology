\outcome{Use iterated integrals to compute multiple integrals.}
\outcome{Apply Fubini's Theorem.}

\begin{document}
\begin{exercise}
  Compute:
  \[
  \int_0^{1}\int_{y}^{1}  \frac{2y}{x^2+y^2}\d x \d y
  \begin{prompt}
    = \answer{\ln(2)}
  \end{prompt}
  \]
  \begin{hint}
    Change the order of integration.
  \end{hint}
\end{exercise}
