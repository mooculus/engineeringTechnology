\begin{exercise}
  Find the maximum and minimum values of $F(x,y,z) =6x+3y+2z$ subject to
  the constraint $4x^2+2y^2+z^2=70$. Write DNE if no such value exists.
  \begin{prompt}
    \begin{align*}
      \text{Maximum value:}&~\answer{35}\\
      \text{Minimum value:}&~\answer{-35}
    \end{align*}
  \end{prompt}
\end{exercise}
