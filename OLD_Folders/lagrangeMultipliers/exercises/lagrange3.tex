\begin{exercise}
  Find three positive numbers whose sum is $48$ and whose product is
  as large as possible.
  \begin{prompt}
    Write your answer $a$, $b$, $c$, in order with $a\le b\le c$.
    \begin{align*}
      a &= \answer{16}\\
      b &= \answer{16}\\
      c &= \answer{16}
    \end{align*}
  \end{prompt}
\end{exercise}
