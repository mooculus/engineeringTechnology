\outcome{Work in polar coordinates.}
\outcome{Compute double integrals in polar coordinates.}

\begin{document}
\begin{exercise}
  When evaluating
  \[
  \iint_R F(x,y)\d A
  \]
  using polar coordinates, $F(x,y)$ is replaced with
  $F\left(\answer{r\cos(\theta)},\answer{r\sin(\theta)}\right)$ and
  $\d A$ is replaced with $\answer{r d r d\theta}$.
\end{exercise}
