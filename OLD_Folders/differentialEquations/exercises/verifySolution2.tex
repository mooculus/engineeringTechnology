\begin{exercise}
Consider the differential equation $\frac{dy}{dx} -2y =0$.

What is meant by a solution to this equation?

\begin{multipleChoice}
\choice{The solution is an $x$-value for which the above relationship holds.}
\choice[correct]{The solution is a function $y(x)$ for which the above relationship holds for all values of $x$ in its domain.}
\end{multipleChoice}



Select all of the functions below that are solutions to this equation.

\begin{selectAll}
\choice[correct]{$y = 0$}
\choice{$y = 2x$}
\choice[correct]{$y = 4e^{2x}$}
\choice{$y = 4e^{2x}+C$}
\choice{$y = 7e^{-2x}$}
\choice{$y = 3\sin(2x)$}
\end{selectAll}


\end{exercise}
