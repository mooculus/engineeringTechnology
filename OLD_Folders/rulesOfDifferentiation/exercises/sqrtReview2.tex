\begin{exercise}

Select the correct statement below.

\begin{multipleChoice}
\choice{$\frac{\frac{a+\sqrt{x^3}}{\sqrt{x}}}{\frac{\sqrt{a+x^3}}{x}} = \frac{\sqrt{x}}{x} = \frac{1}{\sqrt{x}}$}

\choice{$\frac{\frac{a+\sqrt{x^3}}{\sqrt{x}}}{\frac{\sqrt{a+x^3}}{x}} = \frac{x\left(a+\sqrt{x^3}\right)}{\sqrt{x}\sqrt{a+x^3}} = \frac{x}{\sqrt{x}} = \sqrt{x}$}

\choice{$\frac{\frac{a+\sqrt{x^3}}{\sqrt{x}}}{\frac{\sqrt{a+x^3}}{x}} = \frac{x\left(a+\sqrt{x^3}\right)\sqrt{a+x^3}}{\sqrt{x}} = \sqrt{x}\left(a+\sqrt{x^3}\right)\sqrt{a+x^3}$}

\choice{$\frac{\frac{a+\sqrt{x^3}}{\sqrt{x}}}{\frac{\sqrt{a+x^3}}{x}} = \frac{\left(a+\sqrt{x^3}\right)\sqrt{a+x^3}}{x\sqrt{x}}$}

\choice[correct]{$\frac{\frac{a+\sqrt{x^3}}{\sqrt{x}}}{\frac{\sqrt{a+x^3}}{x}} = \frac{x\left(a+\sqrt{x^3}\right)}{\sqrt{x}\sqrt{a+x^3}} = \frac{ax + \sqrt{x^5}}{\sqrt{ax + x^4}}$}

\choice{$\frac{\frac{a+\sqrt{x^3}}{\sqrt{x}}}{\frac{\sqrt{a+x^3}}{x}} = \frac{x\left(a+\sqrt{x^3}\right)}{\sqrt{x}\sqrt{a+x^3}} = \frac{ax + \sqrt{x^4}}{\sqrt{ax + x^4}}$}

\choice{$\frac{\frac{a+\sqrt{x^3}}{\sqrt{x}}}{\frac{\sqrt{a+x^3}}{x}} = \frac{x\left(a+\sqrt{x^3}\right)}{\sqrt{x}\sqrt{a+x^3}} = \frac{ax + \sqrt{x^5}}{x\left(a+x^3\right)}$}

\end{multipleChoice}

\end{exercise}
