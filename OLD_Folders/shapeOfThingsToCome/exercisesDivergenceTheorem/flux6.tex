\begin{exercise}
  Let $\vec{F}(x,y,z) = \vector{z^3,x^2y,y^2z}$ and
  \[
  R = \{(x,y,z): x^2+y^2+z\le 4, 0\le z\le 1\}.
  \]
  Compute the flux of $\vec{F}$ across the surface of $R$.
  \begin{prompt}
  \[
  \oiint_{\partial R} \vec{F}\dotp\uvec{n} \d S = \answer{37\pi/6}
  \]
  \end{prompt}
\end{exercise}
