\begin{exercise}
  Let $\vec{F}(x,y,z) = \vector{x^3+\cos (z),\sin (x)+y^3,x^3+y^3}$ and
  \[
  R = \{(x,y,z): x^2+y^2\le 4, -1\le z \le 3\}.
  \]
  Compute the flux of $\vec{F}$ across the surface of $R$.
  \begin{prompt}
  \[
  \oiint_{\partial R} \vec{F}\dotp\uvec{n} \d S = \answer{96\pi}
  \]
  \end{prompt}
\end{exercise}
