\begin{exercise}
  Let $C$ be a circle (drawn in a counterclockwise fashion when viewed
  from  a height of $z=10$) of radius $1$ centered at $(0,0,3)$ in the plane $2x-3y+z =
  3$.  Compute:
  \[
  \oint_C (y-z) \d x + (x+y^2)\d y + (y-z^2) \d z
  \begin{prompt}
    = \answer{5\pi/\sqrt{14}}
  \end{prompt}
  \]
\end{exercise}
