\begin{exercise}
  Let
  \[
  H = \{(x,y,z): x^2+y^2+z^2=4,z\ge 0 \}
  \]
  with an upward orientation, and let $\vec{F}(x,y,z) =
  \vector{z-2y-x^2,x-y^2+z, x+y}$. Compute:
  \[
  \iint_H (\curl\vec{F})\dotp\uvec{n} \d S
  \begin{prompt}
    = \answer{12\pi}
  \end{prompt}
  \]
\end{exercise}
