\begin{exercise}
  Let $T$ be a triangle (drawn in a counterclockwise fashion when
  viewed from $z=10$) with vertices:
  \[
  (2,4,1)\quad (2,2,1) \quad (5,3,2)
  \]
  Compute:
  \[
  \oint_T (z-x) \d x + (x+y+z)\d y + (z^2-x) \d z
  \begin{prompt}
    = \answer{4}
  \end{prompt}
  \]
\end{exercise}
