\begin{exercise}
  Let
  \[
  H = \{(x,y,z): x^2+y^2+z^2=9,x\ge 0 \}
  \]
  oriented in the positive $x$-direction, and let $\vec{F}(x,y,z) =
  \vector{x-y-z,y^2-x,y-z^2}$. Compute:
  \[
  \iint_H (\curl\vec{F})\dotp\uvec{n} \d S
  \begin{prompt}
    = \answer{9\pi}
  \end{prompt}
  \]
\end{exercise}
