\begin{exercise}
 
Consider the function $F(x,y) = 4x^2-y^2+2$.  A representation of several curves are given below.  Select all that are a part of a level curve of $F(x,y)$.

\begin{selectAll}
\choice{$y=x$}
\choice[correct]{$y=2x$}
\choice{$y=4x$}
\choice[correct]{$4x^2-y^2=14$}
\choice{$\vec{r}(t) =\vector{t,4t+1}$}
\choice[correct]{$\vec{r}(t) =\vector{t^2,2t^2}$}
\choice{$\vec{r}(t) =\vector{t^2,2t^2,2}$}
\end{selectAll}  

\end{exercise}
