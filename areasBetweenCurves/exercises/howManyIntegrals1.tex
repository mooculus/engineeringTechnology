\outcome{Set up an integral with respect to $x$.}

\begin{document}
\begin{exercise}
	Consider the region bounded by $y=e^x, y=2e^{-x}+1$, and $x=0$.
	\begin{image}
	\begin{tikzpicture}
		\begin{axis}[
			domain=-0.5:2.5, ymax=3.5,xmax=.9, ymin=-0.25, xmin=-0.25,
			axis lines =center, xlabel=$x$, ylabel=$y$,
            		every axis y label/.style={at=(current axis.above origin),anchor=south},
            		every axis x label/.style={at=(current axis.right of origin),anchor=west},
            		axis on top,
            		]
                      
            	\addplot [draw=penColor,very thick,smooth,samples=800] {exp(x)};
            	\addplot [draw=penColor2,very thick,smooth] {2*exp(-x)+1};
	
	
		\addplot [name path=A,domain=0:ln(2),draw=none] {exp(x)};   
            	\addplot [name path=B,domain=0:ln(2),draw=none] {2*exp(-x)+1};
		\addplot [fillp] fill between[of=A and B];
		
		\node at (axis cs:0.3,1.1) [penColor] {$y=e^{x}$};
            	\node at (axis cs:0.5,2.7) [penColor2] {$y=2e^{-x}+1$};
                       
            	\end{axis}
	\end{tikzpicture}
	\end{image}
	
	How many integrals with respect to $x$ are needed to compute this area? $\answer{1}$
		
	How many integrals with respect to $y$ are needed to compute this area? $\answer{2}$
\end{exercise}
