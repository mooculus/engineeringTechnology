\begin{exercise}
	Consider the region bounded above by $y=4\sqrt{2x}$ and below by $y=4x^2$ and $y=-4x+6$ in the first quadrant.
	\begin{image}
	\begin{tikzpicture}
		\begin{axis}[
			domain=-1:3.5, ymax=9,xmax=2.5, ymin=-0.25, xmin=-0.25,
			axis lines =center, xlabel=$x$, ylabel=$y$,
            		every axis y label/.style={at=(current axis.above origin),anchor=south},
            		every axis x label/.style={at=(current axis.right of origin),anchor=west},
            		axis on top,
            		]
                      
            	\addplot [draw=penColor,very thick,smooth,samples=800] {4*sqrt(2*x)};
            	\addplot [draw=penColor2,very thick,smooth] {2*(x^2)};
		\addplot [draw=black!60!green,very thick,smooth] {-4*x+6};
	
	
		\addplot [name path=A,domain=1/2:1,draw=none,samples=500] {4*sqrt(2*x)};   
            	\addplot [name path=B,domain=1/2:1,draw=none] {-4*x+6};
		\addplot [name path=C,domain=0.99:2,draw=none] {4*sqrt(2*x)};
		\addplot [name path=D,domain=0.99:2,draw=none] {2*(x^2)};
		\addplot [fillp] fill between[of=A and B];
		\addplot [fillp] fill between[of=C and D];
		
		\node at (axis cs:1,7) [penColor] {$y=4\sqrt{2x}$};
            	\node at (axis cs:1.95,5) [penColor2] {$y=4x^2$};
		\node at (axis cs:1.65,1.75) [black!60!green] {$y=-4x+6$};
                       
            	\end{axis}
	\end{tikzpicture}
	\end{image}
	
	How many integrals with respect to $x$ are needed to compute this area? $\answer{2}$
		
	How many integrals with respect to $y$ are needed to compute this area? $\answer{2}$
\end{exercise}
