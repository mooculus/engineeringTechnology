\begin{exercise}

Let $h(x) = f(g(x))$, where $f$ and $g$ are functions which have the properties shown in the table below.  Evaluate $h'(4)$.

\[
\begin{array}{c|c|c|c|c}
 x & f(x) & f'(x) & g(x) & g'(x)\\ \hline
1 & -9 & -4 & 0 & 6\\
2 & 1 & -4 & -7 & -9\\
3 & 4 & 3 & 7 & 0\\
4 & -3 & -3 & 1 & -5\\
5 & 2 & 6 & 4 & 8\\
6 & 8 & -9 & 5 & 2\\
7 & -8 & -3 & -2 & -4\\
8 & 2 & 3 & 0 & 8\\
9 & 8 & -1 & 7 & -6\\
10 & -5 & 6 & -7 & -2\\
\end{array}
\]


\begin{prompt}
	$$h'(4) = \answer{20}$$
\end{prompt}


\end{exercise}
