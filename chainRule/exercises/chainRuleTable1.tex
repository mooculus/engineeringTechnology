\begin{exercise}

Let $f(x) = g(x^2)$, where $g$ is a function which has the properties shown in the table below.  Evaluate $f'(3)$.

\[
\begin{array}{c|c|c}
 x & g(x) & g'(x)\\ \hline
1 & 3 & 1\\
2 & 7 & -2\\
3 & -5 & -3\\
4 & -8 & -7\\
5 & 2 & 7\\
6 & 6 & -2\\
7 & 1 & -9\\
8 & 3 & -1\\
9 & 6 & 4\\
10 & 8 & 1\\
\end{array}
\]

\begin{prompt}
	$$f'(3) = \answer{24}$$
\end{prompt}


\end{exercise}
