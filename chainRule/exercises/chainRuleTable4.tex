\begin{exercise}

Let $f(x) = \frac{1}{g(x^2)}$, where $g$ is a function which has the properties shown in the table below.  Evaluate $f'(3)$.

\[
\begin{array}{c|c|c}
 x & g(x) & g'(x)\\ \hline
1 & 6 & 6\\
2 & 3 & -9\\
3 & 9 & -8\\
4 & -9 & -9\\
5 & 3 & -9\\
6 & 5 & 2\\
7 & 8 & -7\\
8 & 3 & 8\\
9 & -3 & 5\\
10 & 9 & -9\\
\end{array}
\]

\begin{prompt}
	$$f'(3) = \answer{\frac{-10}{3}}$$
\end{prompt}


\end{exercise}
