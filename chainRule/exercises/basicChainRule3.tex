\begin{exercise}
Let $s(n) = \sec {\left(\frac{5^{4}}{n^{3}}\right)}$. Compute:

\[
\dd{n}s(n)
\begin{prompt}=\answer{-3}\cdot\frac{5^{4}}{n^{\answer{4}}}\sec {\left(\answer{\frac{5^{4}}{n^{3}}}\right)}\cdot \answer{\tan} {\left(\frac{5^{4}}{n^{3}}\right)}\end{prompt}
\]
\end{exercise}
