\begin{exercise}
Let $a(k) = \sin \left(3 \sqrt[4]{2} \sqrt[4]{k}\right)$. Compute:

\[
\dd{k}a(k)
\begin{prompt}=\answer{\frac{3 \cos \left(3 2^{1/4} k^{1/4}\right)}{2^{7/4} k^{3/4}}}\end{prompt}
\]
\end{exercise}
