\begin{exercise}

Suppose $\theta$ is an angle such that $\sec\theta = \sqrt{37}$ and $\cot\theta < 0$.

Based on this data, $\theta$ lies in quadrant \wordChoice{\choice{I}\choice{II}\choice{III}\choice[correct]{IV}}.

The sign of $\sin\theta$ is \wordChoice{\choice{positive}\choice[correct]{negative}}.

Select the value of $\left|\sin\theta\right|$ below.

\begin{multipleChoice}
\choice{$\frac{1}{\sqrt{37}}$}
\choice{$\frac{1}{\sqrt{38}}$}
\choice{$\sqrt{37}$}
\choice[correct]{$\frac{6}{\sqrt{37}}$}
\choice{$\frac{\sqrt{38}}{\sqrt{37}}$}
\choice{$\sqrt{38}$}
\end{multipleChoice}

The sign of $\tan\theta$ is \wordChoice{\choice{positive}\choice[correct]{negative}}.

Select the value of $\left|\tan\theta\right|$ below.

\begin{multipleChoice}
\choice{$\sqrt{38}$}
\choice{$\frac{1}{\sqrt{38}}$}
\choice[correct]{$6$}
\choice{$\frac{1}{\sqrt{37}}$}
\choice{$\frac{1}{6}$}
\choice{$\sqrt{37}$}
\end{multipleChoice}

\end{exercise}
