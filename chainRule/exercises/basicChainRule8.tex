\begin{exercise}
Let $g(u) = \csc (2 \cos (2 u))$. Compute:

\[
\dd{u}g(u)
\begin{prompt}=\answer{4 \sin (2 u) \cot (2 \cos (2 u)) \csc (2 \cos (2 u))}\end{prompt}
\]
\end{exercise}
