          (-1,2)   (3,3)   (4,1)  (4,5)  (-1,2) (3,3)};
      \end{axis}
    \end{tikzpicture}
  \end{image}
  Let $\partial Q$ be the boundary of $Q$ drawn in a counterclockwise
  fashion. Compute:
  \[
  \oint_{\partial Q} \left( (x^2 e^{(x^2)} -3y)\d x + \sin(y^2)\d y\right)
  \begin{prompt}
    = \answer{16.5}
  \end{prompt}
  \]
