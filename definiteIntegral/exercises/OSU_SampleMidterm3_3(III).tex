\outcome{Derive the derivatives of inverse trig functions.}

\outcome{Understand the relationship between indefinite and definite integrals.}

\begin{document}
\begin{exercise}

Determine the indefinite integral $\int\left(\frac{1}{1+x^2}-3\right)dx$.

\begin{prompt}
\begin{multipleChoice}
\choice{$\ln(1+x^2)+C$}
\choice{$\ln(1+x^2)-3x+C$}
\choice[correct]{$\tan^{-1}(x)-3x+C$}
\choice{$\tan^{-1}(x)$}
\choice{$\tan^{-1}(x)+C$}
\choice{None of the previous answers}
\end{multipleChoice}
\end{prompt}
\end{exercise}
