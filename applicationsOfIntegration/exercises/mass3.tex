\outcome{Use integrals to compute mass of a wire with variable density}

\begin{document}
\begin{exercise}
	Find the mass of a length of wire that goes from $x=0$ to $x=2\pi$ and whose density is given by $\rho(x) =|\cos(x)|.$

The mass is $\answer{4}$ units.

\end{exercise}
