\begin{document}
\begin{exercise}

Consider 
\[
f(x) = \begin{cases} 
	x^2+x &\text{if  $x<3$,}\\
	2x^2-5x+9 &\text{if $x \geq 3$.}
\end{cases}
\]


Determine if $f$ is differentiable at $x=3$, and if it is, find the value of its derivative there.

\begin{prompt}
	If $f$ is differentiable, give its value below.  Otherwise, write "DNE".
	
	$f'(3) = \answer{7}$
\end{prompt}

\end{exercise}
