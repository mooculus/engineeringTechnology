\begin{exercise}

The graph of a function $f$ is given below.


\begin{image}
  \begin{tikzpicture}
    \begin{axis}[
        xmin=-0.3,xmax=4.3,ymin=-0.3,ymax=4.3,
        clip=true,
        unit vector ratio*=1 1 1,
        axis lines=center,
        grid = major,
        ytick={0,1,...,4},
    xtick={0,1,...,4},
        xlabel=$x$, ylabel=$y$,
        every axis y label/.style={at=(current axis.above origin),anchor=south},
        every axis x label/.style={at=(current axis.right of origin),anchor=west},
      ]
      \addplot[very thick,penColor,domain=-0.3:4.3,samples=50] plot{x^3/8};
      
      \node at (axis cs:1.5,2.5) [black] {$y=f(x)$};
      \end{axis}`
  \end{tikzpicture}
\end{image}

Select the correct statement in each of the sets below.

\begin{multipleChoice}
\choice{$f(1) = f(3)$}
\choice{$f(1) > f(3)$}
\choice[correct]{$f(1) < f(3)$}
\end{multipleChoice}

\begin{multipleChoice}
\choice{$f'(1) = f'(3)$}
\choice{$f'(1) > f'(3)$}
\choice[correct]{$f'(1) < f'(3)$}
\end{multipleChoice}

Solve the equation $f(c) = 1$.

\begin{multipleChoice}
\choice{$c = 1$}
\choice[correct]{$c = 2$}
\choice{$c = 3$}
\choice{$c = 4$}
\end{multipleChoice}

\end{exercise}
