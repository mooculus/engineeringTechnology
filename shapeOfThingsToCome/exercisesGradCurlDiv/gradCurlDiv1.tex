In the exercises that follow, consider the function $F : \R^2 \to \R$ defined by the rule $F(x,y) = 3y - x^3$.

\begin{exercise}
  Compute \(\divergence F\).
  \begin{multipleChoice}
    \choice{This is zero.}
    \choice{This can be computed (and is nonzero).}
    \choice[correct]{That does not even make sense.}
  \end{multipleChoice}
\end{exercise}

\begin{exercise}

  \end{multipleChoice}
\end{exercise}

\begin{exercise}
  Compute the gradient \(\grad F\).

  \end{multipleChoice}
  \begin{exercise}
    \[
      \grad F = \langle\answer{ -3  x^2 },\answer{ 3 }\rangle

\end{exercise}
