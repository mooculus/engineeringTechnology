Which of the following statements must be true?  Mark all that apply.
\begin{selectAll}
	\choice[correct]{$f$ has an absolute maximum on $\mathbb{R}$}
	\choice{$f$ has an absolute minimum on $\mathbb{R}$}
	\choice[correct]{$f(b) < \frac{f(a)+f(c)}{2}$}
	\choice{The absolute max of $f$ occurs at $x=c$}
	\choice[correct]{$f$ has a local max at $x=a$}
\end{selectAll}
