\begin{exercise}


A sign chart for the derivative of a function $f$ is pictured below.  $f$ has domain $[-1.5,3)$, and is continuous on that domain.  Each dashed line indicates a place where $f'(x)=0$:

\begin{image}
\begin{tikzpicture}
	\begin{axis}[
            trim axis left,
            scale only axis,
            domain=-1:4,
            ymax=2,
            ymin=-2,
            axis lines=none,
            height=3cm, %% Hard coded height! 
            width=\textwidth, %% width
          ]
          \addplot [draw=none, fill=fill1, domain=(-1.5:0)] {2} \closedcycle;
          \addplot [draw=none, fill=fill2, domain=(0:1)] {2} \closedcycle;
          \addplot [draw=none, fill=fill1, domain=(1:3)] {2} \closedcycle;
          
          \addplot [->,textColor] plot coordinates {(-2,0) (4,0)}; %% axis{0};
          
          \addplot [dashed, textColor] plot coordinates {(0,0) (0,2)};
          \addplot [dashed, textColor] plot coordinates {(1,0) (1,2)};

          
          \node at (axis cs:-1.5,0) [anchor=north,textColor] {\footnotesize$-1.5$};
          \node at (axis cs:0,0) [anchor=north,textColor] {\footnotesize$0$};
          \node at (axis cs:1,0) [anchor=north,textColor] {\footnotesize$1$};
          \node at (axis cs:3,0) [anchor=north,textColor] {\footnotesize$3$};

          \node at (axis cs:-0.75,1) [textColor] {\footnotesize$f'(x)>0$};
          \node at (axis cs:0.5,1) [textColor] {\footnotesize$f'(x)<0$};
          \node at (axis cs:2,1) [textColor] {\footnotesize$f'(x)>0$};


        \end{axis}
\end{tikzpicture}
\end{image}



Which of the following statements must be true?  Mark all that apply.
\begin{selectAll}
	\choice{$f$ has an absolute maximum on $[-1.5,3)$}
	\choice[correct]{$f$ has an absolute minimum on $[-1.5,3)$}
	\choice[correct]{For all $x \in (0,2)$, $f(x) \geq f(1)$}
	\choice{The absolute min of $f$ occurs at $x=1$}
	\choice{$f$ has a local max at $x=1$}
\end{selectAll}
