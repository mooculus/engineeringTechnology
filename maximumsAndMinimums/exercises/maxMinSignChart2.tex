A sign chart for the derivative of a function $f$ is pictured below.  Each dashed line indicates a place where $f'(x)=0$:

\begin{image}
\begin{tikzpicture}
	\begin{axis}[
            trim axis left,
            scale only axis,
            domain=-3:3,
            ymax=2,
            ymin=-2,
            axis lines=none,
            height=3cm, %% Hard coded height! 
            width=\textwidth, %% width
          ]
          \addplot [draw=none, fill=fill2, domain=(-3:-1)] {2} \closedcycle;
          \addplot [draw=none, fill=fill2, domain=(-1:1)] {2} \closedcycle;
          \addplot [draw=none, fill=fill1, domain=(1:3)] {2} \closedcycle;
          
          \addplot [->,textColor] plot coordinates {(-3,0) (3,0)}; %% axis{0};
          
          \addplot [dashed, textColor] plot coordinates {(-1,0) (-1,2)};
          \addplot [dashed, textColor] plot coordinates {(1,0) (1,2)};

          
          \node at (axis cs:-1,0) [anchor=north,textColor] {\footnotesize$-1$};
          \node at (axis cs:1,0) [anchor=north,textColor] {\footnotesize$1$};

          \node at (axis cs:-2,1) [textColor] {\footnotesize$f'(x)<0$};
          \node at (axis cs:0,1) [textColor] {\footnotesize$f'(x)<0$};
          \node at (axis cs:2,1) [textColor] {\footnotesize$f'(x)>0$};

        \end{axis}
\end{tikzpicture}
\end{image}

On the interval $[-2,0]$,  $f$ is 
\begin{multipleChoice}
	\choice{Increasing}
	\choice[correct]{Decreasing}
	\choice{Not monotonic}
	\choice{Cannot tell}
\end{multipleChoice}

At $x=-1$, $f$ has a 
\begin{multipleChoice}
	\choice{Local max}
	\choice{Local min}
	\choice[correct]{Neither}
	\choice{Cannot tell}
\end{multipleChoice}

Which of the following statements is true?
\begin{multipleChoice}
	\choice{For all $x \in (0,2)$, $f(x) \leq f(1)$}
	\choice{For all $x \in (0,2)$, $f(x) = f(1)$}
	\choice[correct]{For all $x \in (0,2)$, $f(x) \geq f(1)$}
\end{multipleChoice}
