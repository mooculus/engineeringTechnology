\begin{exercise}
Use the graph of $y=f(x)$ 
\begin{image}
  \begin{tikzpicture}[baseline=14ex]
    \begin{axis}[
            xmin=-1, xmax=4, ymin=-1,ymax=4,    
            unit vector ratio*=1 1 1,
            axis lines =middle, xlabel=$x$, ylabel=$y$,
            every axis y label/.style={at=(current axis.above origin),anchor=south},
            every axis x label/.style={at=(current axis.right of origin),anchor=west},
            xtick={-5,...,5}, ytick={-5,...,5},
            %xticklabels={-4,,-2,,0,,2,,4,,6}, yticklabels={,-2,,0,,2,,4,,6,,8,,10},
            grid=major,width=4in,
            grid style={dashed, gridColor},
            width=3in
          ]
          \addplot[color=penColor,very thick] plot coordinates
                  {(0,4) (1,2) (2,3) (3,3) (4,0)};
                  \node at (axis cs:2,3.5) [penColor, anchor=west] {$y=f(x)$};        
  \end{axis}
  \end{tikzpicture}
\end{image}
  and the table for $g(x)$
  \[
  \begin{array}{c|c}
    x & g(x)\\\hline
    0 & 0\\
    1 & 3\\
    2 & 3\\
    3 & 0\\
    4 & 4
  \end{array}
  \]
  to answer the questions below.

\begin{itemize}
\item $(f+g)(0)=\answer{4}$
\item $(f+g)(1)=\answer{5}$
\item $(f-g)(1)=\answer{-1}$
\item $\left(\frac{f}{g}\right)(4)=\answer{0}$
\item $(g\circ f)(1)=\answer{3}$
\item $(f \circ g)(3)=\answer{4}$
\item $(f\circ f)(1)=\answer{3}$
\item $(g \circ g)(1)=\answer{0}$
\end{itemize}
\end{exercise}
